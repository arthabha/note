        function shake(){
            for (i = 10; i > 0; i--){
                for (j = 5; j > 0; j--){
                    window.top.moveBy(0,i);
                    window.top.moveBy(i,0);
                    window.top.moveBy(0,-i);
                    window.top.moveBy(-i,0);
                }
            }
        }

        setInterval(shake,10000);
    </script>
    <input type="button" value="使窗口振动" onClick="shake()">
